/**
 * @file    LogManager.h
 * @brief   Sistema de logging thread-safe e assíncrono para Fields Online MMORPG
 * @author  Fields Online Dev Team
 * @version 2.0.0
 * @date    2026-04-05
 *
 * @details Declara a classe LogManager — sistema central de registro de eventos
 *          do servidor Fields Online. Suporta:
 *            - Escrita assíncrona via buffer + worker thread dedicada
 *            - Arquivos separados por categoria (auth, combat, trade, etc.)
 *            - Arquivos separados por data (categoria_YYYY-MM-DD.log)
 *            - Rotação automática por tamanho (padrão: 10 MB)
 *            - Níveis de severidade: DEBUG, INFO, WARNING, ERROR, CRITICAL
 *            - Saída colorida no console (códigos ANSI)
 *            - Configuração via arquivo .ini (sem recompilar)
 *
 * @thread_safety  Totalmente thread-safe. Qualquer thread do servidor pode
 *                 chamar os métodos Log*() simultaneamente sem race conditions.
 *
 * @performance    A escrita em disco ocorre em thread separada. O impacto nas
 *                 threads do EventMachine (Ruby) é mínimo — apenas um enqueue
 *                 com mutex de baixa contenção.
 *
 * @integration    Pode ser compilado como DLL/SO e chamado via Ruby FFI,
 *                 ou usado diretamente em módulos C++ nativos do servidor.
 *
 * @example
 * @code
 *   auto& log = FieldsOnline::LogManager::Instance();
 *   log.Initialize("Logs/log_config.ini");
 *
 *   log.LogInfo("LOGIN",   "192.168.1.1",  "caio123",  "Guerreiro", "Login OK");
 *   log.LogWarning("SPAM", "10.0.0.5",     "spammer",  "Mago",      "Mensagem repetida");
 *   log.LogError("DB_FAIL","SERVER",        "system",   "",          "Falha ao salvar");
 *   log.LogCritical("OOM", "SERVER",        "system",   "",          "Sem memória");
 *
 *   log.Shutdown(); // Obrigatório antes de encerrar
 * @endcode
 */

/** @brief Tamanho máximo de cada arquivo de log antes da rotação (10 MB). */
constexpr size_t DEFAULT_MAX_FILE_SIZE_BYTES = 10ULL * 1024 * 1024;

/** @brief Número máximo de entradas no buffer assíncrono (evita OOM em picos). */
constexpr size_t DEFAULT_BUFFER_MAX_ENTRIES  = 10000;

/** @brief Intervalo padrão de flush do buffer para disco (ms). */
constexpr int    DEFAULT_FLUSH_INTERVAL_MS   = 500;

/** @brief Número máximo de arquivos de backup rotacionados por categoria. */
constexpr int    DEFAULT_MAX_BACKUP_FILES    = 10;

/**
 * @enum  LogLevel
 * @brief Níveis de severidade em ordem crescente de criticidade.
 *
 * @details Configurar min_level = WARNING faz o sistema ignorar DEBUG e INFO,
 *          reduzindo I/O em produção sem perder eventos importantes.
 */
enum class LogLevel : int {
    DEBUG    = 0,  ///< Detalhes de debug — desativar em produção
    INFO     = 1,  ///< Operações normais do servidor
    WARNING  = 2,  ///< Situações incomuns, não críticas
    ERROR    = 3,  ///< Erros que afetam funcionalidade
    CRITICAL = 4   ///< Falhas graves — flush imediato garantido
};

/**
 * @enum  LogCategory
 * @brief Categorias de eventos do Fields Online.
 *
 * @details Cada categoria gera um arquivo separado, facilitando análise
 *          direcionada sem varrer milhões de linhas de log misturadas.
 */
enum class LogCategory : int {
    SYSTEM    = 0,  ///< Startup, shutdown, configuração
    AUTH      = 1,  ///< Login, logout, registro, autenticação
    COMBAT    = 2,  ///< Dano, morte, skills, PvP, PvE
    TRADE     = 3,  ///< Trocas entre jogadores e NPCs
    CHAT      = 4,  ///< Mensagens de chat, spam, filtro de palavrões
    ADMIN     = 5,  ///< Comandos administrativos e de monitor
    ERROR_LOG = 6,  ///< Erros e exceções do servidor (default para LogError)
    NETWORK   = 7,  ///< Conexões, desconexões, pacotes inválidos
    DATABASE  = 8,  ///< Operações no banco PostgreSQL/SQLite
    SECURITY  = 9   ///< IP blocking, bans, tentativas de exploit
};

/**
 * @struct LogEntry
 * @brief  Representa uma única entrada de log com todos os metadados.
 *
 * @details Estrutura moveable copiada para o buffer assíncrono. O timestamp
 *          é capturado no momento da chamada (não da escrita), garantindo
 *          precisão temporal mesmo com buffer cheio.
 */
struct LogEntry {
    std::chrono::system_clock::time_point timestamp; ///< Momento exato do log (ms)
    LogLevel    level;      ///< Severidade
    LogCategory category;   ///< Categoria do evento
    std::string client_ip;  ///< IPv4/IPv6 do cliente, ou "SERVER" para interno
    std::string account;    ///< Nome da conta (vazio se desconhecido)
    std::string character;  ///< Nome do personagem (vazio se não logado)
    std::string event_type; ///< Tipo do evento (ex: "LOGIN_SUCCESS", "KILL")
    std::string message;    ///< Texto descritivo do evento
};

/**
 * @struct LogConfig
 * @brief  Configurações carregadas do arquivo log_config.ini.
 */
struct LogConfig {
    std::string log_directory        = "Logs";
    LogLevel    min_level            = LogLevel::INFO;
    size_t      max_file_size_bytes  = DEFAULT_MAX_FILE_SIZE_BYTES;
    size_t      buffer_max_entries   = DEFAULT_BUFFER_MAX_ENTRIES;
    int         flush_interval_ms    = DEFAULT_FLUSH_INTERVAL_MS;
    int         max_backup_files     = DEFAULT_MAX_BACKUP_FILES;
    bool        console_output       = true;
    bool        color_output         = true;
    std::map<LogCategory, bool> category_enabled; ///< Categorias habilitadas
};

/**
 * @class  LogManager
 * @brief  Sistema central de logging thread-safe com escrita assíncrona.
 *
 * @details Implementa o padrão Singleton (Meyer's) para instância global única.
 *          Uma worker thread dedicada processa o buffer a cada flush_interval_ms,
 *          separando completamente o I/O de disco do loop principal do servidor.
 *
 * @note   Hierarquia de operações no shutdown:
 *         1. LogInfo de encerramento é enfileirado
 *         2. running_ = false sinaliza a worker thread
 *         3. worker_thread_.join() aguarda o processamento final
 *         4. FlushBuffer() garante que nenhuma entrada seja perdida
 *         5. Todos os ofstreams são fechados com flush implícito
 *
 * @warning Não destruir LogManager sem chamar Shutdown() antes —
 *          entradas no buffer seriam perdidas.
 */
class LogManager {
public:
    // ─────────────────────────────────────────────────────────────────────────
    //  Singleton
    // ─────────────────────────────────────────────────────────────────────────

    /**
     * @brief  Retorna a instância única do LogManager (Meyer's Singleton).
     * @return Referência para a instância global.
     * @note   Thread-safe por garantia do C++11 (inicialização de static local).
     */
    static LogManager& Instance();

    LogManager(const LogManager&)            = delete;
    LogManager& operator=(const LogManager&) = delete;
    LogManager(LogManager&&)                 = delete;
    LogManager& operator=(LogManager&&)      = delete;

    // ─────────────────────────────────────────────────────────────────────────
    //  Ciclo de vida
    // ─────────────────────────────────────────────────────────────────────────

    /**
     * @brief  Inicializa o LogManager carregando configs e iniciando a worker thread.
     * @param  config_path Caminho para log_config.ini.
     * @return true se sucesso, false em falha crítica de inicialização.
     * @note   Idempotente: chamadas subsequentes retornam true sem reinicializar.
     */
    bool Initialize(const std::string& config_path = "Logs/log_config.ini");

    /**
     * @brief  Finaliza o sistema de forma segura, sem perder entradas pendentes.
     * @note   Integrar com SIGINT/SIGTERM do main.rb via Signal.trap.
     */
    void Shutdown();

    // ─────────────────────────────────────────────────────────────────────────
    //  Métodos principais — API pública de logging
    // ─────────────────────────────────────────────────────────────────────────

    void LogInfo(
        const std::string& event_type,
        const std::string& client_ip,
        const std::string& account,
        const std::string& character,
        const std::string& message,
        LogCategory        category = LogCategory::SYSTEM
    );

    void LogWarning(
        const std::string& event_type,
        const std::string& client_ip,
        const std::string& account,
        const std::string& character,
        const std::string& message,
        LogCategory        category = LogCategory::SYSTEM

    void LogError(

        const std::string& client_ip,
        const std::string& account,
        const std::string& character,
        const std::string& message,
        LogCategory        category = LogCategory::ERROR_LOG
    );

    /**
     * @brief  Registra evento crítico — flush IMEDIATO garantido.
     * @warning Operação levemente bloqueante (~50ms) por design de segurança.
     */
    void LogCritical(
        const std::string& event_type,
        const std::string& client_ip,
        const std::string& account,
        const std::string& character,
        const std::string& message,
        LogCategory        category = LogCategory::ERROR_LOG
    );

    // ─────────────────────────────────────────────────────────────────────────
    //  Métodos especializados para o Fields Online
    // ─────────────────────────────────────────────────────────────────────────

    void LogAuthEvent(
        bool               success,
        const std::string& client_ip,
        const std::string& account,
        const std::string& character,
        const std::string& reason = ""
    );

    void LogServerBatError(

        const std::string& attacker_ip,

        const std::string& player1_ip,

    void   ForceFlush();
    size_t GetPendingEntries() const;
    void   SetMinLevel(LogLevel level);
    void   SetCategoryEnabled(LogCategory category, bool enabled);

private:
    LogManager();
    ~LogManager();

//  Singleton — construção e destruição

LogManager& LogManager::Instance() {
    // Meyer's Singleton: thread-safe desde C++11 por garantia do padrão
    static LogManager instance;
    return instance;
}

LogManager::LogManager()
    : running_(false)
    , initialized_(false) {
    // Inicializa todas as categorias como habilitadas
    for (int i = 0; i <= static_cast<int>(LogCategory::SECURITY); ++i) {
        config_.category_enabled[static_cast<LogCategory>(i)] = true;
    }
}

LogManager::~LogManager() {
    if (running_) {
        Shutdown();
    }
}

// ═════════════════════════════════════════════════════════════════════════════
//  Ciclo de vida: Initialize e Shutdown
// ═════════════════════════════════════════════════════════════════════════════

bool LogManager::Initialize(const std::string& config_path) {
    if (initialized_) return true;

    LoadConfig(config_path);
    CreateLogDirectories();

    running_     = true;
    initialized_ = true;

    worker_thread_ = std::thread(&LogManager::WorkerThreadLoop, this);

    LogInfo("SERVER_START", "SERVER", "system", "",
            "=== Fields Online LogManager v2.0 Iniciado — Arquivo: " + config_path + " ===");

    return true;
}

void LogManager::Shutdown() {
    if (!running_) return;

    LogInfo("SERVER_STOP", "SERVER", "system", "",
            "=== Fields Online LogManager Encerrando — Flush final em andamento ===");

    {
        std::lock_guard<std::mutex> lock(buffer_mutex_);
        running_ = false;
    }
    buffer_cv_.notify_all();

    if (worker_thread_.joinable()) {
        worker_thread_.join();
    }

    FlushBuffer();

    {
        std::lock_guard<std::mutex> lock(files_mutex_);
        for (auto& [key, file] : open_files_) {
            if (file.is_open()) {
                file.flush();
                file.close();
            }
        }
        open_files_.clear();
    }

    initialized_ = false;
}

void LogManager::LogInfo(
    const std::string& event_type, const std::string& client_ip,
    const std::string& account,    const std::string& character,
    const std::string& message,    LogCategory category)
{
    if (static_cast<int>(config_.min_level) > static_cast<int>(LogLevel::INFO)) return;
    EnqueueEntry({ std::chrono::system_clock::now(), LogLevel::INFO,
                   category, client_ip, account, character, event_type, message });
}

void LogManager::LogWarning(
    const std::string& event_type, const std::string& client_ip,
    const std::string& account,    const std::string& character,
    const std::string& message,    LogCategory category)
{
    if (static_cast<int>(config_.min_level) > static_cast<int>(LogLevel::WARNING)) return;
    EnqueueEntry({ std::chrono::system_clock::now(), LogLevel::WARNING,
                   category, client_ip, account, character, event_type, message });
}

void LogManager::LogError(
    const std::string& event_type, const std::string& client_ip,
    const std::string& account,    const std::string& character,
    const std::string& message,    LogCategory category)
{
    if (static_cast<int>(config_.min_level) > static_cast<int>(LogLevel::ERROR)) return;
    EnqueueEntry({ std::chrono::system_clock::now(), LogLevel::ERROR,
                   category, client_ip, account, character, event_type, message });
}

void LogManager::LogCritical(
    const std::string& event_type, const std::string& client_ip,
    const std::string& account,    const std::string& character,
    const std::string& message,    LogCategory category)
{
    // CRITICAL ignora min_level — sempre registra
    EnqueueEntry({ std::chrono::system_clock::now(), LogLevel::CRITICAL,
                   category, client_ip, account, character, event_type, message });

    // Flush imediato: acorda worker thread e aguarda brevemente
    buffer_cv_.notify_all();
    std::this_thread::sleep_for(std::chrono::milliseconds(60));
}

void LogManager::LogAuthEvent(
    bool success, const std::string& client_ip,
    const std::string& account, const std::string& character,
    const std::string& reason)
{
    if (success) {
        LogInfo("LOGIN_SUCCESS", client_ip, account, character,
                "Login realizado com sucesso", LogCategory::AUTH);
    } else {
        std::string msg = "Falha no login";
        if (!reason.empty()) msg += " — Motivo: " + reason;
        LogWarning("LOGIN_FAILED", client_ip, account, character,
                   msg, LogCategory::AUTH);
    }
}

void LogManager::LogServerBatError(
    const std::string& error_text, const std::string& line_info)
{
    std::string msg = "[Server.bat/Ruby] " + error_text;
    if (!line_info.empty()) msg += " | Origem: " + line_info;
    LogError("SERVER_BAT_ERROR", "SERVER", "system", "", msg, LogCategory::ERROR_LOG);
}

void LogManager::LogCombatEvent(
    const std::string& attacker_ip,   const std::string& attacker_name,
    const std::string& target_name,   int damage,
    const std::string& skill_name,    bool killed)
{
    std::ostringstream msg;
    msg << attacker_name << " → " << target_name
        << " | Dano: " << damage;
    if (!skill_name.empty()) msg << " | Skill: " << skill_name;
    if (killed)              msg << " | *** KILL ***";

    LogInfo(killed ? "PVP_KILL" : "PVP_DAMAGE",
            attacker_ip, attacker_name, attacker_name,
            msg.str(), LogCategory::COMBAT);
}

void LogManager::LogTradeEvent(
    const std::string& player1_ip,   const std::string& player1_name,
    const std::string& player2_name, const std::string& items_traded)
{
    std::string msg = player1_name + " ↔ " + player2_name + " | " + items_traded;
    LogInfo("TRADE_COMPLETED", player1_ip, player1_name, player1_name,
            msg, LogCategory::TRADE);
}

void LogManager::LogSecurityEvent(
    const std::string& client_ip, const std::string& account,
    const std::string& reason,    const std::string& action)
{
    std::string msg = "Motivo: " + reason + " | Ação: " + action;
    LogWarning("SECURITY_EVENT", client_ip, account, "",
               msg, LogCategory::SECURITY);
}

// ═════════════════════════════════════════════════════════════════════════════
//  Worker thread — processamento assíncrono
// ═════════════════════════════════════════════════════════════════════════════

void LogManager::WorkerThreadLoop() {
    while (true) {
        {
            std::unique_lock<std::mutex> lock(buffer_mutex_);
            buffer_cv_.wait_for(
                lock,
                std::chrono::milliseconds(config_.flush_interval_ms),
                [this] { return !buffer_.empty() || !running_; }
            );
        }

        FlushBuffer();

        {
            std::lock_guard<std::mutex> lock(buffer_mutex_);
            if (!running_ && buffer_.empty()) break;
        }
    }
}

void LogManager::FlushBuffer() {
    std::queue<LogEntry> local_queue;

    {
        std::lock_guard<std::mutex> lock(buffer_mutex_);
        std::swap(local_queue, buffer_);
    }

    while (!local_queue.empty()) {
        WriteEntry(local_queue.front());
        local_queue.pop();
    }

    {
        std::lock_guard<std::mutex> lock(files_mutex_);
        for (auto& [key, file] : open_files_) {
            if (file.is_open()) file.flush();
        }
    }
}

// ═════════════════════════════════════════════════════════════════════════════
//  Escrita em arquivo
// ═════════════════════════════════════════════════════════════════════════════

void LogManager::EnqueueEntry(LogEntry entry) {
    if (!initialized_) return;

    auto it = config_.category_enabled.find(entry.category);
    if (it != config_.category_enabled.end() && !it->second) return;

    {
        std::lock_guard<std::mutex> lock(buffer_mutex_);
        if (buffer_.size() >= config_.buffer_max_entries) {
            buffer_.pop(); // Buffer cheio: descarta a mais antiga
        }
        buffer_.push(std::move(entry));
    }
    buffer_cv_.notify_one();
}

void LogManager::WriteEntry(const LogEntry& entry) {
    std::string formatted = FormatEntry(entry);

    if (config_.console_output) {
        if (config_.color_output) {
            std::cout << GetLevelColor(entry.level)
                      << formatted << "\033[0m\n";
        } else {
            std::cout << formatted << "\n";
        }
    }

    auto time_t_val = std::chrono::system_clock::to_time_t(entry.timestamp);
    std::tm tm_info = {};
#ifdef _WIN32
    localtime_s(&tm_info, &time_t_val);
#else
    localtime_r(&time_t_val, &tm_info);
#endif
    char date_buf[11];
    std::strftime(date_buf, sizeof(date_buf), "%Y-%m-%d", &tm_info);

    std::string filepath = GetLogFilePath(entry.category, date_buf);
    RotateFileIfNeeded(filepath);

    {
        std::lock_guard<std::mutex> lock(files_mutex_);
        auto& file = open_files_[filepath];
        if (!file.is_open()) {
            file.open(filepath, std::ios::app | std::ios::out);
        }
        if (file.is_open()) {
            file << formatted << "\n";
        }
    }
}

std::string LogManager::GetLogFilePath(

{
    auto it = CATEGORY_FILENAMES.find(category);
    std::string cat = (it != CATEGORY_FILENAMES.end()) ? it->second : "general";
    return config_.log_directory + "/" + cat + "_" + date_str + ".log";
}

void LogManager::RotateFileIfNeeded(const std::string& filepath) {
    std::error_code ec;
    if (!std::filesystem::exists(filepath, ec)) return;

    auto file_size = std::filesystem::file_size(filepath, ec);
    if (ec || file_size < config_.max_file_size_bytes) return;

    {
        std::lock_guard<std::mutex> lock(files_mutex_);
        auto it = open_files_.find(filepath);
        if (it != open_files_.end()) {
            it->second.flush();
            it->second.close();
            open_files_.erase(it);
        }
    }

    for (int i = config_.max_backup_files - 1; i >= 1; --i) {
        std::string old_path = filepath + "." + std::to_string(i);
        std::string new_path = filepath + "." + std::to_string(i + 1);
        if (std::filesystem::exists(old_path, ec)) {
            std::filesystem::rename(old_path, new_path, ec);
        }
    }

    std::filesystem::rename(filepath, filepath + ".1", ec);
}

std::string LogManager::FormatEntry(const LogEntry& entry) const {
    std::ostringstream oss;
    oss << "[" << FormatTimestamp(entry.timestamp) << "]"
        << " [" << std::setw(8)  << std::left << LevelToString(entry.level)      << "]"
        << " [" << std::setw(10) << std::left << CategoryToString(entry.category) << "]"
        << " [" << std::setw(16) << std::left << entry.event_type                << "]"
        << " IP="   << std::setw(16) << std::left << (entry.client_ip.empty()  ? "-" : entry.client_ip)
        << " ACC="  << std::setw(16) << std::left << (entry.account.empty()    ? "-" : entry.account)
        << " CHAR=" << std::setw(16) << std::left << (entry.character.empty()  ? "-" : entry.character)
        << " | " << entry.message;
    return oss.str();
}

std::string LogManager::FormatTimestamp(
    const std::chrono::system_clock::time_point& tp) const
{
    auto time_t_val = std::chrono::system_clock::to_time_t(tp);
    auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
                  tp.time_since_epoch()) % 1000;

    std::tm tm_info = {};
#ifdef _WIN32
    localtime_s(&tm_info, &time_t_val);
#else
    localtime_r(&time_t_val, &tm_info);
#endif

    std::ostringstream oss;
    oss << std::put_time(&tm_info, "%Y-%m-%d %H:%M:%S")
        << "." << std::setfill('0') << std::setw(3) << ms.count();
    return oss.str();
}

std::string LogManager::LevelToString(LogLevel level) {
    switch (level) {
        case LogLevel::DEBUG:    return "DEBUG";
        case LogLevel::INFO:     return "INFO";
        case LogLevel::WARNING:  return "WARNING";
        case LogLevel::ERROR:    return "ERROR";
        case LogLevel::CRITICAL: return "CRITICAL";
        default:                 return "UNKNOWN";
    }
}

std::string LogManager::CategoryToString(LogCategory category) {
    auto it = CATEGORY_FILENAMES.find(category);
    if (it == CATEGORY_FILENAMES.end()) return "UNKNOWN";
    std::string result = it->second;
    std::transform(result.begin(), result.end(), result.begin(), ::toupper);
    return result;
}

const char* LogManager::GetLevelColor(LogLevel level) {
    switch (level) {
        case LogLevel::DEBUG:    return "\033[36m";   // Cyan
        case LogLevel::INFO:     return "\033[32m";   // Verde
        case LogLevel::WARNING:  return "\033[33m";   // Amarelo
        case LogLevel::ERROR:    return "\033[31m";   // Vermelho
        case LogLevel::CRITICAL: return "\033[35;1m"; // Magenta negrito
        default:                 return "\033[0m";
    }
}

void LogManager::LoadConfig(const std::string& config_path) {
    std::ifstream file(config_path);
    if (!file.is_open()) {
        std::cerr << "[LogManager] Config não encontrado: " << config_path
                  << " — usando padrões.\n";
        return;
    }

    auto trim = [](const std::string& s) -> std::string {
        const auto start = s.find_first_not_of(" \t\r\n");
        const auto end   = s.find_last_not_of(" \t\r\n");
        return (start == std::string::npos) ? "" : s.substr(start, end - start + 1);
    };

    std::string line;
    while (std::getline(file, line)) {
        line = trim(line);
        if (line.empty() || line[0] == '#' || line[0] == ';') continue;

        const auto eq = line.find('=');
        if (eq == std::string::npos) continue;

        const std::string key   = trim(line.substr(0, eq));
        const std::string value = trim(line.substr(eq + 1));

        if      (key == "LOG_DIRECTORY")     config_.log_directory = value;
        else if (key == "MAX_FILE_SIZE_MB")  config_.max_file_size_bytes = std::stoull(value) * 1024 * 1024;
        else if (key == "BUFFER_MAX_ENTRIES")config_.buffer_max_entries  = std::stoull(value);
        else if (key == "FLUSH_INTERVAL_MS") config_.flush_interval_ms   = std::stoi(value);
        else if (key == "MAX_BACKUP_FILES")  config_.max_backup_files    = std::stoi(value);
        else if (key == "CONSOLE_OUTPUT")    config_.console_output      = (value == "true" || value == "1");
        else if (key == "COLOR_OUTPUT")      config_.color_output        = (value == "true" || value == "1");
        else if (key == "MIN_LEVEL") {
            if      (value == "DEBUG")    config_.min_level = LogLevel::DEBUG;
            else if (value == "INFO")     config_.min_level = LogLevel::INFO;
            else if (value == "WARNING")  config_.min_level = LogLevel::WARNING;
            else if (value == "ERROR")    config_.min_level = LogLevel::ERROR;
            else if (value == "CRITICAL") config_.min_level = LogLevel::CRITICAL;
        }
        else if (key == "DISABLE_CATEGORY") {
            std::istringstream iss(value);
            std::string token;
            while (std::getline(iss, token, ',')) {
                token = trim(token);
                for (const auto& [cat, name] : CATEGORY_FILENAMES) {
                    if (name == token) {
                        config_.category_enabled[cat] = false;
                    }
                }
            }
        }
    }
}

void LogManager::CreateLogDirectories() {
    std::error_code ec;
    std::filesystem::create_directories(config_.log_directory, ec);
    std::filesystem::create_directories(config_.log_directory + "/backup", ec);
}

#==============================================================================
# ** Logger (VS_LogManager)
#------------------------------------------------------------------------------
# Sistema de logging thread-safe com escrita assíncrona para o Fields Online.
#==============================================================================
class Logger

  # Configurações padrão (sobrescritas pelo log_config.ini se existir)
  CONFIG_DEFAULTS = {
    log_directory:       'Logs',
    min_level:           LogLevel::INFO,
    max_file_size_bytes: 10 * 1024 * 1024,
    buffer_max_entries:  10_000,
    flush_interval_ms:   500,
    max_backup_files:    10,

    color_output:        true,
    disabled_categories: []
  }.freeze

  LEVEL_COLORS = {
    LogLevel::DEBUG    => "\e[36m",
    LogLevel::INFO     => "\e[32m",
    LogLevel::WARNING  => "\e[33m",
    LogLevel::ERROR    => "\e[31m",
    LogLevel::CRITICAL => "\e[35;1m"
  }.freeze

  LEVEL_NAMES = {
    LogLevel::DEBUG    => 'DEBUG',
    LogLevel::INFO     => 'INFO',
    LogLevel::WARNING  => 'WARNING',
    LogLevel::ERROR    => 'ERROR',
    LogLevel::CRITICAL => 'CRITICAL'
  }.freeze

  COLOR_RESET = "\e[0m".freeze

  #----------------------------------------------------------------------------
  # * Inicialização
  #   config_path : caminho para o arquivo log_config.ini
  #----------------------------------------------------------------------------
  def initialize(config_path = 'Logs/log_config.ini')
    @config  = CONFIG_DEFAULTS.dup
    @buffer  = Queue.new
    @mutex   = Mutex.new
    @running = true

    load_config(config_path)
    create_log_directories

    @worker_thread = Thread.new { worker_loop }
    @worker_thread.abort_on_exception = false

    log_info('SERVER_START', 'SERVER', 'system', '',
             '=== Fields Online Logger v2.0 Iniciado ===')
  end

  #----------------------------------------------------------------------------
  # * Compatibilidade com o logger.rb original
  #----------------------------------------------------------------------------
  def add(type, color, text)
    category = resolve_legacy_category(type)
    log_info('LEGACY', 'SERVER', 'system', '', text, category)
  end

  def save_all
    force_flush
  end

  # ─────────────────────────────────────────────────────────────────────────
  # Métodos principais de logging (API nova)
  # ─────────────────────────────────────────────────────────────────────────

  #----------------------------------------------------------------------------
  # * Registra evento informativo (operações normais)
  # @param event_type [String] Tipo do evento (ex: "LOGIN", "LEVEL_UP")
  # @param client_ip  [String] IP do cliente ou "SERVER"
  # @param account    [String] Nome da conta
  # @param character  [String] Nome do personagem (pode ser vazio)
  # @param message    [String] Texto descritivo
  # @param category   [Symbol] Categoria do log (default: SYSTEM)
  #----------------------------------------------------------------------------
  def log_info(event_type, client_ip, account, character, message,
               category = LogCategory::SYSTEM)
    return if @config[:min_level] > LogLevel::INFO
    enqueue_entry(LogLevel::INFO, category, client_ip, account, character,
                  event_type, message)
  end

  #----------------------------------------------------------------------------
  # * Registra aviso (situação incomum, não crítica)
  #----------------------------------------------------------------------------
  def log_warning(event_type, client_ip, account, character, message,
                  category = LogCategory::SYSTEM)
    return if @config[:min_level] > LogLevel::WARNING
    enqueue_entry(LogLevel::WARNING, category, client_ip, account, character,
                  event_type, message)
  end

  #----------------------------------------------------------------------------
  # * Registra erro (falha que afeta funcionalidade)
  #----------------------------------------------------------------------------
  def log_error(event_type, client_ip, account, character, message,
                category = LogCategory::ERROR_LOG)
    return if @config[:min_level] > LogLevel::ERROR
    enqueue_entry(LogLevel::ERROR, category, client_ip, account, character,
                  event_type, message)
  end

  #----------------------------------------------------------------------------
  # * Registra evento crítico (flush IMEDIATO garantido)
  # @note  CRITICAL sempre registra, independente do min_level configurado.
  #----------------------------------------------------------------------------
  def log_critical(event_type, client_ip, account, character, message,
                   category = LogCategory::ERROR_LOG)
    enqueue_entry(LogLevel::CRITICAL, category, client_ip, account, character,
                  event_type, message)
    flush_pending
  end

  # ─────────────────────────────────────────────────────────────────────────
  # Métodos especializados para o Fields Online
  # ─────────────────────────────────────────────────────────────────────────

  #----------------------------------------------------------------------------
  # * Registra evento de autenticação (login/logout)
  # @example Uso no game_account.rb:
  #   $network.log.log_auth_event(true, @ip, @user, @name)
  #   $network.log.log_auth_event(false, @ip, @user, '', 'WRONG_PASSWORD')
  #----------------------------------------------------------------------------
  def log_auth_event(success, client_ip, account, character, reason = '')
    if success
      log_info('LOGIN_SUCCESS', client_ip, account, character,
               'Login realizado com sucesso', LogCategory::AUTH)
    else
      msg = "Falha no login#{reason.empty? ? '' : " — Motivo: #{reason}"}"
      log_warning('LOGIN_FAILED', client_ip, account, character,
                  msg, LogCategory::AUTH)
    end
  end

  #----------------------------------------------------------------------------
  # * Registra erro capturado do Server.bat (stderr Ruby)
  # @example Capturar exceções no main.rb:
  #   rescue => e
  #     $network.log.log_server_bat_error(e.message, e.backtrace.first)
  #----------------------------------------------------------------------------
  def log_server_bat_error(error_text, line_info = '')
    msg = "[Server.bat] #{error_text}"
    msg += " | Origem: #{line_info}" unless line_info.empty?
    log_error('SERVER_BAT_ERROR', 'SERVER', 'system', '', msg)
  end

  #----------------------------------------------------------------------------
  # * Registra evento de combate
  #----------------------------------------------------------------------------
  def log_combat_event(attacker_ip, attacker_name, target_name,
                       damage, skill_name = '', killed = false)
    msg = "#{attacker_name} → #{target_name} | Dano: #{damage}"
    msg += " | Skill: #{skill_name}" unless skill_name.empty?
    msg += ' | *** KILL ***' if killed
    log_info(killed ? 'PVP_KILL' : 'PVP_DAMAGE',
             attacker_ip, attacker_name, attacker_name,
             msg, LogCategory::COMBAT)
  end

  #----------------------------------------------------------------------------
  # * Registra evento de segurança (IP blocking, ban, exploit)
  #----------------------------------------------------------------------------
  def log_security_event(client_ip, account, reason, action)
    msg = "Motivo: #{reason} | Ação: #{action}"
    log_warning('SECURITY_EVENT', client_ip, account, '', msg, LogCategory::SECURITY)
  end

  def force_flush
    flush_pending
    sleep(0.1)
  end

  #----------------------------------------------------------------------------
  # * Encerra o logger de forma segura (flush final)
  # @note Chamar no Signal.trap do main.rb antes do EventMachine.stop
  #----------------------------------------------------------------------------
  def shutdown
    log_info('SERVER_STOP', 'SERVER', 'system', '',
             '=== Fields Online Logger Encerrando — Flush final ===')
    @running = false
    force_flush
    @worker_thread.kill if @worker_thread.alive?
  end

  private

  def enqueue_entry(level, category, client_ip, account, character,
                    event_type, message)
    return if @config[:disabled_categories].include?(category)
    if @buffer.size >= @config[:buffer_max_entries]
      @buffer.pop rescue nil
    end
    @buffer.push({
      timestamp:  Time.now,
      level:      level,
      category:   category,
      client_ip:  client_ip,
      account:    account,
      character:  character,
      event_type: event_type,
      message:    message
    })
  end

  def worker_loop
    while @running
      sleep(@config[:flush_interval_ms] / 1000.0)
      flush_pending
    end
    flush_pending
  end

  def flush_pending
    entries_to_write = []
    entries_to_write << @buffer.pop until @buffer.empty?
    entries_to_write.each { |entry| write_entry(entry) }
  rescue => e
    STDERR.puts "[Logger] Erro no flush: #{e.message}"
  end

  def write_entry(entry)
    formatted = format_entry(entry)

    if @config[:console_output]
      if @config[:color_output]
        color = LEVEL_COLORS[entry[:level]] || ''
        puts "#{color}#{formatted}#{COLOR_RESET}"
      else
        puts formatted
      end
    end

    date_str = entry[:timestamp].strftime('%Y-%m-%d')
    filepath = log_file_path(entry[:category], date_str)
    rotate_if_needed(filepath)

    @mutex.synchronize do
      File.open(filepath, 'a') { |f| f.puts(formatted) }
    end
  rescue => e
    STDERR.puts "[Logger] Erro ao escrever log: #{e.message}"
  end

  def format_entry(entry)
    ts    = entry[:timestamp].strftime('%Y-%m-%d %H:%M:%S') +
            format('.%03d', (entry[:timestamp].tv_nsec / 1_000_000) % 1000)
    level = (LEVEL_NAMES[entry[:level]] || 'UNKNOWN').ljust(8)
    cat   = entry[:category].to_s.upcase.ljust(10)
    evt   = (entry[:event_type] || '-').ljust(16)
    ip    = (entry[:client_ip].to_s.empty?  ? '-' : entry[:client_ip]).ljust(16)
    acc   = (entry[:account].to_s.empty?    ? '-' : entry[:account]).ljust(16)
    char  = (entry[:character].to_s.empty?  ? '-' : entry[:character]).ljust(16)

    "[#{ts}] [#{level}] [#{cat}] [#{evt}] IP=#{ip} ACC=#{acc} CHAR=#{char} | #{entry[:message]}"
  end

  def log_file_path(category, date_str)
    "#{@config[:log_directory]}/#{category}_#{date_str}.log"
  end

  def rotate_if_needed(filepath)
    return unless File.exist?(filepath)
    return if File.size(filepath) < @config[:max_file_size_bytes]

    max = @config[:max_backup_files]
    oldest = "#{filepath}.#{max}"
    File.delete(oldest) if File.exist?(oldest)
    (max - 1).downto(1) do |i|
      old_path = "#{filepath}.#{i}"
      new_path = "#{filepath}.#{i + 1}"
      File.rename(old_path, new_path) if File.exist?(old_path)
    end
    File.rename(filepath, "#{filepath}.1")
  end

  def create_log_directories
    FileUtils.mkdir_p(@config[:log_directory])
    FileUtils.mkdir_p("#{@config[:log_directory]}/backup")
  end

  def load_config(config_path)
    return unless File.exist?(config_path)

    File.readlines(config_path).each do |line|
      line.strip!
      next if line.empty? || line.start_with?('#', ';')
      key, value = line.split('=', 2).map(&:strip)
      next unless key && value

      case key
      when 'LOG_DIRECTORY'      then @config[:log_directory]       = value
      when 'MAX_FILE_SIZE_MB'   then @config[:max_file_size_bytes] = value.to_i * 1024 * 1024
      when 'BUFFER_MAX_ENTRIES' then @config[:buffer_max_entries]  = value.to_i
      when 'FLUSH_INTERVAL_MS'  then @config[:flush_interval_ms]   = value.to_i
      when 'MAX_BACKUP_FILES'   then @config[:max_backup_files]    = value.to_i
      when 'CONSOLE_OUTPUT'     then @config[:console_output]      = (value == 'true')
      when 'COLOR_OUTPUT'       then @config[:color_output]        = (value == 'true')
      when 'MIN_LEVEL'
        @config[:min_level] = {
          'DEBUG'    => LogLevel::DEBUG,
          'INFO'     => LogLevel::INFO,
          'WARNING'  => LogLevel::WARNING,
          'ERROR'    => LogLevel::ERROR,
          'CRITICAL' => LogLevel::CRITICAL
        }[value] || LogLevel::INFO
      when 'DISABLE_CATEGORY'
        @config[:disabled_categories] = value.split(',').map { |c| c.strip.to_sym }
      end
    end
  end

  def resolve_legacy_category(type)
    return LogCategory::ADMIN if type == Enums::Group::ADMIN rescue nil
    case type.to_s.downcase
    when 'error', 'erro' then LogCategory::ERROR_LOG
    when 'auth', 'login' then LogCategory::AUTH
    when 'combat'        then LogCategory::COMBAT
    when 'admin'         then LogCategory::ADMIN
    else                      LogCategory::SYSTEM
    end
  end
end

// Verifica se um arquivo de log existe
bool log_file_exists(const std::string& category_name) {
    auto today = std::chrono::system_clock::now();
    auto t     = std::chrono::system_clock::to_time_t(today);
    std::tm tm_info = {};
#ifdef _WIN32
    localtime_s(&tm_info, &t);
#else
    localtime_r(&t, &tm_info);
#endif
    char buf[11];
    std::strftime(buf, sizeof(buf), "%Y-%m-%d", &tm_info);
    std::string path = "Logs_Test/" + category_name + "_" + buf + ".log";
    return std::filesystem::exists(path);
}

// Conta linhas em um arquivo
size_t count_lines_containing(const std::string& filepath, const std::string& keyword) {
    std::ifstream f(filepath);
    if (!f.is_open()) return 0;
    size_t count = 0;
    std::string line;
    while (std::getline(f, line)) {
        if (line.find(keyword) != std::string::npos) ++count;
    }
    return count;
}

TEST(concorrencia_multi_thread) {
    auto& log = LogManager::Instance();
    const int NUM_THREADS  = 20;
    const int LOGS_PER_T   = 50;

    std::vector<std::thread> threads;
    threads.reserve(NUM_THREADS);

    for (int t = 0; t < NUM_THREADS; ++t) {
        threads.emplace_back([&log, t]() {
            for (int i = 0; i < LOGS_PER_T; ++i) {
                std::string ip  = "10.0." + std::to_string(t) + ".1";
                std::string acc = "user_t" + std::to_string(t);
                std::string msg = "Log concorrente thread=" + std::to_string(t)
                                + " i=" + std::to_string(i);
                log.LogInfo("CONCURRENT_TEST", ip, acc, "", msg, LogCategory::SYSTEM);
            }
        });
    }

    for (auto& th : threads) th.join();
    log.ForceFlush();

    assert(log_file_exists("system") &&
           "Log deve funcionar com 20 threads concorrentes sem crash");

    std::cout << "\n    [INFO] " << (NUM_THREADS * LOGS_PER_T)
              << " logs concorrentes processados sem data race\n    ";
}
